main()
{
   int n = 12344321, reverse = 0, temp;
   int result;
   
   temp = n;
   
   while( temp != 0 )
   {
      reverse = reverse * 10;
      reverse = reverse + temp%10;
      temp = temp/10;
   }
 
   if ( n == reverse )
      result = 1;
   else
          result = 0;
           
   return 0;
}
